```python
def fit(self):
    # ============ 初始化阶段 ============
    logger = self.logger
    self.global_steps = 0

    # 训练前验证（可选）
    if self.val_reward_fn is not None and val_before_train:
        val_metrics = self._validate()
        logger.log(val_metrics, step=0)

    # 准备生成配置
    gen_config = GenerationConfig(...)
    generation_manager = LLMGenerationManager(...)

    # ============ 主训练循环 ============
    for epoch in range(total_epochs):
        for batch_dict in train_dataloader:

            # ┌─────────────────────────────────────────────────────────┐
            # │  阶段1: 数据准备与生成 (Rollout)                          │
            # └─────────────────────────────────────────────────────────┘
            batch = DataProto.from_single_dict(batch_dict)
            batch = batch.repeat(repeat_times=n_agent, interleave=True)
            gen_batch = batch.pop(['input_ids', 'attention_mask', 'position_ids'])

            if do_search:
                # 搜索模式：多轮推理-搜索循环
                first_input_ids = gen_batch.batch['input_ids'][:, -max_start_length:]
                final_gen_batch_output = generation_manager.run_llm_loop(
                    gen_batch=gen_batch,
                    initial_input_ids=first_input_ids
                )
                # 计算旧策略log_prob
                output = actor_rollout_wg.compute_log_prob(final_gen_batch_output)
                final_gen_batch_output = final_gen_batch_output.union(output)
            else:
                # 标准模式：直接生成
                gen_batch_output = actor_rollout_wg.generate_sequences(gen_batch)
                final_gen_batch_output = gen_batch_output

            batch = batch.union(final_gen_batch_output)

            # ┌─────────────────────────────────────────────────────────┐
            # │  阶段2: 批次平衡 (Balance Batch)                          │
            # └─────────────────────────────────────────────────────────┘
            self._balance_batch(batch, metrics)  # 按序列长度平衡分布

            # ┌─────────────────────────────────────────────────────────┐
            # │  阶段3: 参考策略计算 (Reference Policy)                   │
            # └─────────────────────────────────────────────────────────┘
            if use_reference_policy:
                ref_log_prob = ref_policy_wg.compute_ref_log_prob(batch)
                batch = batch.union(ref_log_prob)

            # ┌─────────────────────────────────────────────────────────┐
            # │  阶段4: 价值函数计算 (Critic)                             │
            # └─────────────────────────────────────────────────────────┘
            if use_critic:
                values = critic_wg.compute_values(batch)
                batch = batch.union(values)

            # ┌─────────────────────────────────────────────────────────┐
            # │  阶段5: 奖励计算与优势估计 (Reward & Advantage)            │
            # └─────────────────────────────────────────────────────────┘
            if use_rm:
                reward_tensor = rm_wg.compute_rm_score(batch)
                batch = batch.union(reward_tensor)

            # 结合规则奖励
            reward_tensor = reward_fn(batch)
            batch.batch['token_level_scores'] = reward_tensor

            # 应用KL惩罚
            if not use_kl_loss:
                batch, kl_metrics = apply_kl_penalty(batch, kl_ctrl, ...)

            # 计算优势函数
            batch = compute_advantage(batch, adv_estimator='gae/grpo', ...)

            # ┌─────────────────────────────────────────────────────────┐
            # │  阶段6: Critic更新 (Update Critic)                        │
            # └─────────────────────────────────────────────────────────┘
            if use_critic:
                critic_output = critic_wg.update_critic(batch)
                metrics.update(critic_output.meta_info['metrics'])

            # ┌─────────────────────────────────────────────────────────┐
            # │  阶段7: Actor更新 (Update Actor)                          │
            # └─────────────────────────────────────────────────────────┘
            if global_steps >= critic_warmup:
                if do_search and state_masking:
                    batch, metrics = _create_loss_mask(batch, metrics)
                actor_output = actor_rollout_wg.update_actor(batch)
                metrics.update(actor_output.meta_info['metrics'])

            # ┌─────────────────────────────────────────────────────────┐
            # │  阶段8: 验证与保存 (Validation & Checkpoint)               │
            # └─────────────────────────────────────────────────────────┘
            if should_validate:
                val_metrics = self._validate()
                metrics.update(val_metrics)

            if should_save:
                self._save_checkpoint()

            # ┌─────────────────────────────────────────────────────────┐
            # │  阶段9: 日志记录 (Logging)                                │
            # └─────────────────────────────────────────────────────────┘
            metrics.update(compute_data_metrics(batch))
            metrics.update(compute_timing_metrics(batch))
            logger.log(metrics, step=global_steps)

            global_steps += 1

            if global_steps >= total_training_steps:
                return
```

### 各阶段详细解释

**veRL 的 PPO 训练循环是一个经典的强化学习数据流设计**

1. **数据流分离**: 生成阶段、估计阶段、更新阶段三个阶段清晰分离，便于分布式优化
2. **Worker 抽象**: 所有计算密集型操作都委托给 Ray Worker Group，驱动进程只做轻量级协调

#### 阶段1: 数据准备与生成 (rollout)

| 操作 | 说明 |
|------|------|
| `batch.repeat(n_agent)` | 复制数据用于多次rollout，提高样本利用率 |
| `generation_manager.run_llm_loop()` | 多轮推理-搜索交互，返回完整生成轨迹 |
| `compute_log_prob()` | 计算旧策略的 log_prob，用于PPO的 ratio 计算 |

#### 阶段2: 批次平衡


#### 阶段3-5: 估计阶段

| 组件 | 作用 |
|------|------|
| 参考策略 | 计算 `ref_log_prob`，用于KL散度惩罚 |
| Critic | 计算 `values`，用于优势函数估计 |
| 奖励模型 | 计算环境奖励 `token_level_scores` |
| 优势估计 | 使用GAE或GRPO计算 `advantages` 和 `returns` |

```python

# Actor 更新（策略优化）
actor_rollout_wg.update_actor(batch)  # PPO-Clip损失
```

#### 阶段8-9: 辅助操作

| 操作 | 触发条件 |
|------|----------|
| 验证 | 每隔 `test_freq` 步 |
| 保存checkpoint | 每隔 `save_freq` 步 |

### 特别说明：Search-R1 的多轮生成

当 `do_search=True` 时，`generation_manager.run_llm_loop()` 会执行：

```python
# 简化伪代码
def run_llm_loop(gen_batch, initial_input_ids):
    current_input = initial_input_ids

    for turn in range(max_turns):
        # 1. LLM生成
        output = actor_rollout_wg.generate_sequences(current_input)

        # 2. 判断是否需要搜索
        if output.contains_search_query():
            # 3. 调用检索服务
            search_results = retriever.search(output.get_query())

            # 4. 将结果加入输入
            current_input = concatenate(current_input, output, search_results)
        else:
            break  # 生成结束

    return final_output
```

**PPO训练的核心思想**

1. **重要性采样**: 使用 `ratio = π_new/π_old` 避免剧烈策略变化
2. **优势函数**: 衡量某个动作比平均水平"好多少"
3. **KL惩罚**: 防止策略偏离参考策略太远

## 微批次与梯度累积详解

### 核心配置参数

从代码 `verl/workers/actor/dp_actor.py:203-231` 和 `verl/workers/critic/dp_critic.py:146-204` 可以看到：


| 参数 | 含义 |
|------|------|
| `ppo_mini_batch_size` | PPO 策略更新的小批次大小 |
| `ppo_micro_batch_size` | 单次前向传播的微批次大小 |
| `gradient_accumulation` | 梯度累积步数 |

### 两层批处理结构

### 代码实现解析

#### Actor 更新 (`dp_actor.py:203-290`)

```python
def update_policy(self, data: DataProto):
    # 1️⃣ 计算梯度累积步数
    gradient_accumulation = ppo_mini_batch_size // ppo_micro_batch_size

    # 2️⃣ 将完整批次分割成多个 mini-batches (PPO论文建议多次重复使用数据)
    dataloader = batch.split(ppo_mini_batch_size)

    for mini_batch in dataloader:  # 每个 mini-batch 做一次策略更新
        # 3️⃣ 将 mini-batch 分割成多个 micro-batches
        micro_batches = mini_batch.split(ppo_micro_batch_size)

        self.actor_optimizer.zero_grad()

        # 4️⃣ 梯度累积循环
        for micro_batch in micro_batches:
            # 前向传播
            entropy, log_prob = self._forward_micro_batch(micro_batch)

            # 计算损失
            pg_loss, pg_clipfrac, ppo_kl = compute_policy_loss(...)
            policy_loss = pg_loss - entropy_loss * entropy_coeff

            # 关键: 损失除以累积步数
            loss = policy_loss / self.gradient_accumulation

            # 反向传播 (梯度累积)
            loss.backward()

        # 5️⃣ 所有微批次处理完后，执行一次优化器步进
        grad_norm = self._optimizer_step()  # clip_grad_norm + optimizer.step()
```

#### Critic 更新 (`dp_critic.py:146-204`)

```python
def update_critic(self, data: DataProto):
    gradient_accumulation = ppo_mini_batch_size // ppo_micro_batch_size

    dataloader = batch.split(ppo_mini_batch_size)

    for mini_batch in dataloader:
        micro_batches = mini_batch.split(ppo_micro_batch_size)

        self.critic_optimizer.zero_grad()

        for micro_batch in micro_batches:
            # 前向传播
            vpreds = self._forward_micro_batch(micro_batch)

            # 计算 value loss
            vf_loss, vf_clipfrac = compute_value_loss(...)

            # 关键: 损失除以累积步数
            loss = vf_loss / self.gradient_accumulation

            # 反向传播 (梯度累积)
            loss.backward()

        # 优化器步进
        grad_norm = self._optimizer_step()
```

### 动态批次大小

代码还支持动态批次大小 (`use_dynamic_bsz`)：

```python
if use_dynamic_bsz:
    # 按最大token数动态分割微批次
    max_token_len = ppo_max_token_len_per_gpu * ulysses_sequence_parallel_size
    micro_batches, _ = rearrange_micro_batches(batch=mini_batch, max_token_len=max_token_len)
else:
    micro_batches = mini_batch.split(ppo_micro_batch_size)
```

**为什么需要两层批处理？**

1. **Micro-batch**: 受 GPU 显存限制，确保每次前向传播不会 OOM
2. **Mini-batch**: PPO 论文建议多次重复使用同一批数据进行策略更新，提高样本效率
3. **梯度累积**: 等价于使用更大的批次大小，使梯度估计更稳定

这种设计允许在大模型训练时平衡**显存占用**和**训练稳定性**。

### 与 PPO 论文的关系

PPO 论文中的 **多次 epochs** 对应代码中的 `dataloader = batch.split(ppo_mini_batch_size)` 循环：

```
PPO论文: "We typically use multiple epochs of SGD on the same data"
代码实现: 每个 rollout 批次会被分割成多个 mini-batches，每个 mini-batch 做一次更新
```
